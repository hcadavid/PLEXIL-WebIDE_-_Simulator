Command Stop();

Command pprint(...);

//EXAMPLE

ObstacleAvoid:
{
  Integer steps=0;

  Concurrence{    

    Duration delay1 = Duration ("PT3S"); 
    Duration tolerance = Duration ("PT0.1S");
    Integer currentAngle=0;
    Boolean timingComplete=false;


    StartMoving:{
        MoveForward(30);     
    }
    
    TimedDistance:{
        Repeat true;  
        Wait (delay1), tolerance;            
        pprint ("Time completed");
        timingComplete=true;
    }

    TurnWheelsS:{
        Start (timingComplete==true && currentAngle==30);
        Repeat true;
        timingComplete=false;
        TurnFrontWheels(0);
        currentAngle=0;
        OpenSprinkler();
    }

    TurnWheelsL:{
        Start (timingComplete==true && currentAngle==0);
        Repeat true;
        timingComplete=false;
        TurnFrontWheels(30);
        currentAngle=30;
        OpenSprinkler();
    }

  }
}
